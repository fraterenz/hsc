11/// Description of a birth of a new subclone during proliferation.
12///
13/// Returned by [`Proliferation::proliferate`] whenever a fit variant is drawn
14/// (so the proliferating cell is reassigned to a fresh, previously-empty
15/// subclone). Carries the new clone id, the parent's id, and the hit count
16/// the new clone was tagged with — used by the `--multihits` feature to
17/// resample the new clone's Gillespie rate from the matching fitness tier.
18#[derive(Clone, Copy, Debug)]
19pub struct BirthEvent {
20    pub new_clone_id: CloneId,
21    pub parent_clone_id: CloneId,
22    pub hit_count: HitCount,
23}

26/// Stem cells can either self-renew by symmetric division or differentiate by
27/// asymmetric division.
28pub enum Division {
29    /// A stem cell proliferates and gives rise to a differentiate cell (which
30    /// we discard in our simulations)
31    Asymmetric(Bernoulli),
32    /// A stem cell proliferates and gives to a new stem cell with the same
33    /// genetic material as the mother cell plus some additional mutations
34    /// acquired upon division (fit divisions and neutral ones)
35    #[default]
36    Symmetric,
37}

39#[derive(Clone, Debug, Default)]
40/// All the updates and changes in the system upon proliferation.
41///
42/// The proliferation step is implemented as following:
43///
44/// 1. select the cell `c` that will proliferate next from the clone
45///    with id `reaction` determined by the Gillespie algorithm
46///
47/// 2. draw and assign mb neutral background mutations to `c` from
48///    Poisson(DeltaT * mub), where mub is the backgound mutation rate
49///
50/// 3. draw and assign md neutral division mutations to `c` from
51///    Poisson(mud), where mud is the division mutation rate
52///
53/// 4. draw from a Bernoulli trial a fit mutation and in case assign
54///    `c` to a new clone. This clone must be empty and different from the
55///    old clone which `c` belonged to.
56///
57/// 5. clone the proliferating cell `c` into `c1` and repeat step 3, 4
58///    with `c1` only if we simulated a symmetric division, see [`Division`].
59pub struct Proliferation {
60    pub neutral_mutation: NeutralMutations,
61    pub division: Division,
62}

64impl Proliferation {
65    pub fn new(neutral_mutation: NeutralMutations, division: Division) -> Self {
66        Proliferation {
67            neutral_mutation,
68            division,
69        }
70    }

72    pub fn proliferate(
73        &self,
74        subclones: &mut SubClones,
75        time: f32,
76        proliferating_subclone: CloneId,
77        distributions: &Distributions,
78        rng: &mut impl Rng,
79    ) -> Option<BirthEvent> {
80        //! Sample a random cell from the subclones with id
81        //! `proliferating_subclone`, assign neutral mutations (background and
82        //! those upon proliferation) and finally assign `cell` to a subclone.
83        //!
84        //! For the last step, there are two possible scenarios:
85        //!
86        //! 1. if there is a new fit variant, then `cell` will be assigned to a
87        //!    new **empty** random clone with an id different from `old_subclone_id`
88        //!    and panics if there aren't any empty subclones left. Returns
89        //!    `Some(BirthEvent)` describing the birth.
90        //! 2. else, reassign `cell` to the old subclone with id `old_subclone_id`.
91        //!    Returns `None`.
92        let mut stem_cell = proliferating_cell(subclones, proliferating_subclone, rng);
93        let p = (distributions.u
94            * stem_cell
95                .interdivision_time(time)
96                .with_context(|| "wrong interdivision time")
97                .unwrap()) as f64;
98        // the fit variant is sampled here
99        let clone_id = next_clone(subclones, proliferating_subclone, p, rng);
100        let birth_event = if clone_id != proliferating_subclone {
101            let hit_count = subclones
102                .get_clone(proliferating_subclone)
103                .unwrap()
104                .hit_count()
105                .child();
106            let new_clone = subclones.get_mut_clone_unchecked(clone_id);
107            new_clone.set_parent_id(proliferating_subclone);
108            new_clone.set_hit_count(hit_count);
109            Some(BirthEvent {
110                new_clone_id: clone_id,
111                parent_clone_id: proliferating_subclone,
112                hit_count,
113            })
114        } else {
115            None
116        };
117        debug!("proliferation at time {time}");
118        debug!(
119            "cell with last division time {} is dividing",
120            stem_cell.get_last_division_time()
121        );
122        trace!("cell {stem_cell:#?} is dividing");
123        if let NeutralMutations::UponDivisionAndBackground = self.neutral_mutation {
124            debug!("realise and assign the neutral background mutations to the proliferating cell");
125            self.realise_background_for_cell(&mut stem_cell, time, distributions, rng);
126        }
127        let cells = match self.division {
128            Division::Asymmetric(prob) => {
129                if prob.sample(rng) {
130                    debug!("asymmetric division");
131                    vec![stem_cell]
132                } else {
133                    vec![stem_cell.clone(), stem_cell]
134                }
135            }
136            Division::Symmetric => vec![stem_cell.clone(), stem_cell],
137        };
138        for mut stem_cell in cells {
139            assign_divisional_mutations(&mut stem_cell, &distributions.neutral_poisson, rng);
140            stem_cell
141                .set_last_division_time(time)
142                .with_context(|| "wrong time")
143                .unwrap();
144            subclones
145                .get_mut_clone_unchecked(clone_id)
146                .assign_cell(stem_cell);
147        }
148        birth_event
149    }

151    pub fn realise_background_mutations(
152        &self,
153        subclones: &mut SubClones,
154        time: f32,
155        distributions: &Distributions,
156        rng: &mut impl Rng,
157    ) {
158        //! Force the lazily-deferred background-mutation draw for every cell
159        //! whose `last_division_t` lies strictly before `time`.
160        //!
161        //! Background mutations are normally drawn at the next division (see
162        //! [`Proliferation::proliferate`]); snapshots and phase transitions
163        //! need to read mutation state of all cells, so they call into this
164        //! method to bring every cell up to the current `time`. Cells whose
165        //! last division is at or after `time` are left untouched.
166        //!
167        //! No-op when `self.neutral_mutation` is [`NeutralMutations::UponDivision`].
168        if let NeutralMutations::UponDivisionAndBackground = self.neutral_mutation {
169            debug!("realise and assign the neutral background mutations to all cells");
170            for stem_cell in subclones.get_mut_cells() {
171                self.realise_background_for_cell(stem_cell, time, distributions, rng);
172            }
173        }
174    }
175
176    fn realise_background_for_cell(
177        &self,
178        cell: &mut StemCell,
179        time: f32,
180        distributions: &Distributions,
181        rng: &mut impl Rng,
182    ) {
183        //! Per-cell variant of [`Self::realise_background_mutations`], shared
184        //! and by [`Self::proliferate`] (which operates on a single cell detached
185        //! from [`SubClones`]).
186        if cell.get_last_division_time() < &time {
187            assign_background_mutations(cell, time, &distributions.neutral_poisson, rng);
188        }
189    }
190}

193/// Specifies the kind of neutral mutations to simulate.
194pub enum NeutralMutations {
195    /// Neutral mutations upon division due to errors in the DNA duplication
196    UponDivision,
197    /// Neutral mutations upon division and background mutations (mutations
198    /// appearing during the lifetime of the cell)
199    #[default]
200    UponDivisionAndBackground,
201}

266    #[test]
267    fn proliferate_sets_parent_id_on_new_clone() {
268        // p = u * interdivision_time. With mu near cells and a unit
269        // interdivision time, the Bernoulli draw in next_clone returns true
270        // with overwhelming probability, so within a few iterations at least
271        // one new fit clone is born; its parent_id must equal the proliferating
272        // clone's id, and the wild-type clone must stay parentless.
273        let rng = &mut ChaCha8Rng::seed_from_u64(42);
274        let mut subclones = make_subclones(5, 0.0);
275        let distributions = Distributions::new(Probs::new(50., 50., 99., 0., 100));
276        let proliferation = Proliferation::new(NeutralMutations::UponDivision, Division::Symmetric);
277
278        let mut saw_birth = false;
279        for _ in 0..10 {
280            // Bring all cells back to t=0 so interdivision_time stays at 1.0
281            // each iteration and p stays close to 1.
282            for cell in subclones.get_mut_cells() {
283                cell.set_last_division_time(0.0).unwrap();
284            }
285            let event = proliferation.proliferate(&mut subclones, 1.0, 0, &distributions, rng);
286            let births: Vec<CloneId> = (1..crate::MAX_SUBCLONES as CloneId)
287                .filter(|id| !subclones.get_clone(*id).unwrap().is_empty())
288                .collect();
289            if let Some(&new_id) = births.first() {
290                assert_eq!(
291                    subclones.get_clone(new_id).unwrap().parent_id(),
292                    Some(0_u16),
293                );
294                // BirthEvent must match the observed new clone, and the new
295                // clone's hit_count must be First (parent is WildType).
296                let event = event.expect("proliferate must report a BirthEvent on a fit birth");
297                assert_eq!(event.new_clone_id, new_id);
298                assert_eq!(event.parent_clone_id, 0);
299                assert_eq!(event.hit_count, HitCount::First);
300                assert_eq!(
301                    subclones.get_clone(new_id).unwrap().hit_count(),
302                    HitCount::First,
303                );
304                saw_birth = true;
305                break;
306            } else {
307                // No new clone born -> no event reported.
308                assert!(event.is_none());
309            }
310        }
311        assert!(saw_birth, "no new fit clone born across 10 proliferations");
312        // Wild type stays parentless.
313        assert!(subclones.get_clone(0).unwrap().parent_id().is_none());
314    }

316    #[test]
317    fn proliferate_hit_count_chain_saturates_at_fourth() {
318        // Repeatedly birth new clones starting from the wild type, walking
319        // the hit_count chain WildType -> First -> Second -> Third -> Fourth
320        // and confirming the 5th and later births stay clamped at Fourth.
321        let rng = &mut ChaCha8Rng::seed_from_u64(42);
322        let mut subclones = make_subclones(5, 0.0);
323        let distributions = Distributions::new(Probs::new(50., 50., 99., 0., 100));
324        let proliferation = Proliferation::new(NeutralMutations::UponDivision, Division::Symmetric);
325
326        let expected = [
327            HitCount::First,
328            HitCount::Second,
329            HitCount::Third,
330            HitCount::Fourth,
331            HitCount::Fourth,
332            HitCount::Fourth,
333        ];
334        let mut parent_id: CloneId = 0;
335        for expected_hits in expected {
336            let mut event = None;
337            for _ in 0..100 {
338                for cell in subclones.get_mut_cells() {
339                    cell.set_last_division_time(0.0).unwrap();
340                }
341                if let Some(e) =
342                    proliferation.proliferate(&mut subclones, 1.0, parent_id, &distributions, rng)
343                {
344                    event = Some(e);
345                    break;
346                }
347            }
348            let event = event.expect("expected a birth within 100 tries");
349            assert_eq!(event.parent_clone_id, parent_id);
350            assert_eq!(event.hit_count, expected_hits);
351            assert_eq!(
352                subclones.get_clone(event.new_clone_id).unwrap().hit_count(),
353                expected_hits,
354            );
355            parent_id = event.new_clone_id;
356        }
357    }
