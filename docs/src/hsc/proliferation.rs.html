11/// Stem cells can either self-renew by symmetric division or differentiate by
12/// asymmetric division.
13pub enum Division {
14    /// A stem cell proliferates and gives rise to a differentiate cell (which
15    /// we discard in our simulations)
16    Asymmetric(Bernoulli),
17    /// A stem cell proliferates and gives to a new stem cell with the same
18    /// genetic material as the mother cell plus some additional mutations
19    /// acquired upon division (fit divisions and neutral ones)
20    #[default]
21    Symmetric,
22}

24#[derive(Clone, Debug, Default)]
25/// All the updates and changes in the system upon proliferation.
26///
27/// The proliferation step is implemented as following:
28///
29/// 1. select the cell `c` that will proliferate next from the clone
30///    with id `reaction` determined by the Gillespie algorithm
31///
32/// 2. draw and assign mb neutral background mutations to `c` from
33///    Poisson(DeltaT * mub), where mub is the backgound mutation rate
34///
35/// 3. draw and assign md neutral division mutations to `c` from
36///    Poisson(mud), where mud is the division mutation rate
37///
38/// 4. draw from a Bernoulli trial a fit mutation and in case assign
39///    `c` to a new clone. This clone must be empty and different from the
40///    old clone which `c` belonged to.
41///
42/// 5. clone the proliferating cell `c` into `c1` and repeat step 3, 4
43///    with `c1` only if we simulated a symmetric division, see [`Division`].
44pub struct Proliferation {
45    pub neutral_mutation: NeutralMutations,
46    pub division: Division,
47}
48
49impl Proliferation {
50    pub fn new(neutral_mutation: NeutralMutations, division: Division) -> Self {
51        Proliferation {
52            neutral_mutation,
53            division,
54        }
55    }
56
57    pub fn proliferate(
58        &self,
59        subclones: &mut SubClones,
60        time: f32,
61        proliferating_subclone: CloneId,
62        distributions: &Distributions,
63        rng: &mut impl Rng,
64        verbosity: u8,
65    ) {
66        //! Sample a random cell from the subclones with id
67        //! `proliferating_subclone`, assign neutral mutations (background and
68        //! those upon proliferation) and finally assign `cell` to a subclone.
69        //!
70        //! For the last step, there are two possible scenarios:
71        //!
72        //! 1. if there is a new fit variant, then `cell` will be assigned to a
73        //!    new **empty** random clone with an id different from `old_subclone_id`
74        //!    and panics if there aren't any empty subclones left
75        //! 2. else, reassign `cell` to the old subclone with id `old_subclone_id`
76        let mut stem_cell = proliferating_cell(subclones, proliferating_subclone, verbosity, rng);
77        let p = (distributions.u
78            * stem_cell
79                .interdivision_time(time)
80                .with_context(|| "wrong interdivision time")
81                .unwrap()) as f64;
82        // the fit variant is sampled here
83        let clone_id = next_clone(subclones, proliferating_subclone, p, rng, verbosity);
84        if verbosity > 1 {
85            println!("proliferation at time {time}");
86            println!(
87                "cell with last division time {} is dividing",
88                stem_cell.get_last_division_time()
89            );
90            if verbosity > 2 {
91                println!("cell {stem_cell:#?} is dividing");
92            }
93        }
94        if let NeutralMutations::UponDivisionAndBackground = self.neutral_mutation {
95            assign_background_mutations(
96                &mut stem_cell,
97                time,
98                &distributions.neutral_poisson,
99                rng,
100                verbosity,
101            );
102        }
103        let cells = match self.division {
104            Division::Asymmetric(prob) => {
105                if prob.sample(rng) {
106                    if verbosity > 1 {
107                        println!("asymmetric division");
108                    }
109                    vec![stem_cell]
110                } else {
111                    vec![stem_cell.clone(), stem_cell]
112                }
113            }
114            Division::Symmetric => vec![stem_cell.clone(), stem_cell],
115        };
116        for mut stem_cell in cells {
117            assign_divisional_mutations(
118                &mut stem_cell,
119                &distributions.neutral_poisson,
120                rng,
121                verbosity,
122            );
123            stem_cell
124                .set_last_division_time(time)
125                .with_context(|| "wrong time")
126                .unwrap();
127            subclones
128                .get_mut_clone_unchecked(clone_id)
129                .assign_cell(stem_cell);
130        }
131    }
132}

135/// Specifies the kind of neutral mutations to simulate.
136pub enum NeutralMutations {
137    /// Neutral mutations upon division due to errors in the DNA duplication
138    UponDivision,
139    /// Neutral mutations upon division and background mutations (mutations
140    /// appearing during the lifetime of the cell)
141    #[default]
142    UponDivisionAndBackground,
143}
