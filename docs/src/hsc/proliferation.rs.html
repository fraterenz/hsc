11#[derive(Debug, Clone, Default)]
12/// Stem cells can either self-renew by symmetric division or differentiate by
13/// asymmetric division.
14pub enum Division {
15    /// A stem cell proliferates and gives rise to a differentiate cell (which
16    /// we discard in our simulations)
17    Asymmetric(Bernoulli),
18    /// A stem cell proliferates and gives to a new stem cell with the same
19    /// genetic material as the mother cell plus some additional mutations
20    /// acquired upon division (fit divisions and neutral ones)
21    #[default]
22    Symmetric,
23}
24
25#[derive(Clone, Debug, Default)]
26/// All the updates and changes in the system upon proliferation.
27///
28/// The proliferation step is implemented as following:
29///
30/// 1. select the cell `c` that will proliferate next from the clone
31///    with id `reaction` determined by the Gillespie algorithm
32///
33/// 2. draw and assign mb neutral background mutations to `c` from
34///    Poisson(DeltaT * mub), where mub is the backgound mutation rate
35///
36/// 3. draw and assign md neutral division mutations to `c` from
37///    Poisson(mud), where mud is the division mutation rate
38///
39/// 4. draw from a Bernoulli trial a fit mutation and in case assign
40///    `c` to a new clone. This clone must be empty and different from the
41///    old clone which `c` belonged to.
42///
43/// 5. clone the proliferating cell `c` into `c1` and repeat step 3, 4
44///    with `c1` only if we simulated a symmetric division, see [`Division`].
45pub struct Proliferation {
46    pub neutral_mutation: NeutralMutations,
47    pub division: Division,
48}
49
50impl Proliferation {
51    pub fn new(neutral_mutation: NeutralMutations, division: Division) -> Self {
52        Proliferation {
53            neutral_mutation,
54            division,
55        }
56    }

58    pub fn proliferate(
59        &self,
60        subclones: &mut SubClones,
61        time: f32,
62        proliferating_subclone: CloneId,
63        distributions: &Distributions,
64        rng: &mut impl Rng,
65    ) {
66        //! Sample a random cell from the subclones with id
67        //! `proliferating_subclone`, assign neutral mutations (background and
68        //! those upon proliferation) and finally assign `cell` to a subclone.
69        //!
70        //! For the last step, there are two possible scenarios:
71        //!
72        //! 1. if there is a new fit variant, then `cell` will be assigned to a
73        //!    new **empty** random clone with an id different from `old_subclone_id`
74        //!    and panics if there aren't any empty subclones left
75        //! 2. else, reassign `cell` to the old subclone with id `old_subclone_id`
76        let mut stem_cell = proliferating_cell(subclones, proliferating_subclone, rng);
77        let p = (distributions.u
78            * stem_cell
79                .interdivision_time(time)
80                .with_context(|| "wrong interdivision time")
81                .unwrap()) as f64;
82        // the fit variant is sampled here
83        let clone_id = next_clone(subclones, proliferating_subclone, p, rng);
84        if clone_id != proliferating_subclone {
85            subclones
86                .get_mut_clone_unchecked(clone_id)
87                .set_parent_id(proliferating_subclone);
88        }
89        debug!("proliferation at time {time}");
90        debug!(
91            "cell with last division time {} is dividing",
92            stem_cell.get_last_division_time()
93        );
94        trace!("cell {stem_cell:#?} is dividing");
95        if let NeutralMutations::UponDivisionAndBackground = self.neutral_mutation {
96            debug!("realise and assign the neutral background mutations to the proliferating cell");
97            self.realise_background_for_cell(&mut stem_cell, time, distributions, rng);
98        }
99        let cells = match self.division {
100            Division::Asymmetric(prob) => {
101                if prob.sample(rng) {
102                    debug!("asymmetric division");
103                    vec![stem_cell]
104                } else {
105                    vec![stem_cell.clone(), stem_cell]
106                }
107            }
108            Division::Symmetric => vec![stem_cell.clone(), stem_cell],
109        };
110        for mut stem_cell in cells {
111            assign_divisional_mutations(&mut stem_cell, &distributions.neutral_poisson, rng);
112            stem_cell
113                .set_last_division_time(time)
114                .with_context(|| "wrong time")
115                .unwrap();
116            subclones
117                .get_mut_clone_unchecked(clone_id)
118                .assign_cell(stem_cell);
119        }
120    }

122    pub fn realise_background_mutations(
123        &self,
124        subclones: &mut SubClones,
125        time: f32,
126        distributions: &Distributions,
127        rng: &mut impl Rng,
128    ) {
129        //! Force the lazily-deferred background-mutation draw for every cell
130        //! whose `last_division_t` lies strictly before `time`.
131        //!
132        //! Background mutations are normally drawn at the next division (see
133        //! [`Proliferation::proliferate`]); snapshots and phase transitions
134        //! need to read mutation state of all cells, so they call into this
135        //! method to bring every cell up to the current `time`. Cells whose
136        //! last division is at or after `time` are left untouched.
137        //!
138        //! No-op when `self.neutral_mutation` is [`NeutralMutations::UponDivision`].
139        if let NeutralMutations::UponDivisionAndBackground = self.neutral_mutation {
140            debug!("realise and assign the neutral background mutations to all cells");
141            for stem_cell in subclones.get_mut_cells() {
142                self.realise_background_for_cell(stem_cell, time, distributions, rng);
143            }
144        }
145    }
146
147    fn realise_background_for_cell(
148        &self,
149        cell: &mut StemCell,
150        time: f32,
151        distributions: &Distributions,
152        rng: &mut impl Rng,
153    ) {
154        //! Per-cell variant of [`Self::realise_background_mutations`], shared
155        //! and by [`Self::proliferate`] (which operates on a single cell detached
156        //! from [`SubClones`]).
157        if cell.get_last_division_time() < &time {
158            assign_background_mutations(cell, time, &distributions.neutral_poisson, rng);
159        }
160    }
161}

163#[derive(Clone, Debug, Default)]
164/// Specifies the kind of neutral mutations to simulate.
165pub enum NeutralMutations {
166    /// Neutral mutations upon division due to errors in the DNA duplication
167    UponDivision,
168    /// Neutral mutations upon division and background mutations (mutations
169    /// appearing during the lifetime of the cell)
170    #[default]
171    UponDivisionAndBackground,
172}

186    fn make_subclones(n: usize, last_division_t: f32) -> SubClones {
187        let mut cells = vec![StemCell::new(); n];
188        for cell in &mut cells {
189            cell.set_last_division_time(last_division_t).unwrap();
190        }
191        SubClones::new(cells, n)
192    }

237    #[test]
238    fn proliferate_sets_parent_id_on_new_clone() {
239        // p = u * interdivision_time. With mu near cells and a unit
240        // interdivision time, the Bernoulli draw in next_clone returns true
241        // with overwhelming probability, so within a few iterations at least
242        // one new fit clone is born; its parent_id must equal the proliferating
243        // clone's id, and the wild-type clone must stay parentless.
244        let rng = &mut ChaCha8Rng::seed_from_u64(42);
245        let mut subclones = make_subclones(5, 0.0);
246        let distributions = Distributions::new(Probs::new(50., 50., 99., 0., 100));
247        let proliferation = Proliferation::new(NeutralMutations::UponDivision, Division::Symmetric);
248
249        let mut saw_birth = false;
250        for _ in 0..10 {
251            // Bring all cells back to t=0 so interdivision_time stays at 1.0
252            // each iteration and p stays close to 1.
253            for cell in subclones.get_mut_cells() {
254                cell.set_last_division_time(0.0).unwrap();
255            }
256            proliferation.proliferate(&mut subclones, 1.0, 0, &distributions, rng);
257            let births: Vec<CloneId> = (1..crate::MAX_SUBCLONES as CloneId)
258                .filter(|id| !subclones.get_clone(*id).unwrap().is_empty())
259                .collect();
260            if let Some(&new_id) = births.first() {
261                assert_eq!(
262                    subclones.get_clone(new_id).unwrap().parent_id(),
263                    Some(0_u16),
264                );
265                saw_birth = true;
266                break;
267            }
268        }
269        assert!(saw_birth, "no new fit clone born across 10 proliferations");
270        // Wild type stays parentless.
271        assert!(subclones.get_clone(0).unwrap().parent_id().is_none());
272    }
