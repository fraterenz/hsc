38pub fn switch_to_moran(
39    mut exponential: Exponential,
40    process_options: ProcessOptions,
41    distributions: Distributions,
42    filename: PathBuf,
43    save_sfs_only: bool,
44    save_population: bool,
45    rng: &mut impl Rng,
46) -> Moran {
47    //! End the exponential growing phase and switch to a fixed-size
48    //! population phase.
49    //!
50    //! There is a delay between birth and the end of the exponential phase,
51    //! since stem cells stop dividing exponentially before birth.
52    //! We add background mutations in this interval of time between the
53    //! end of the exponentially growing phase and the Moran process.
54    debug!("switching to Moran at time {}", exponential.time);
55    let (current_time, time_to_restart) = match exponential.phase {
56        ExponentialPhase::Development => (TIME_AT_BIRTH, 0f32),
57        ExponentialPhase::Regrowth => (exponential.time, exponential.time),
58    };
59    if let NeutralMutations::UponDivisionAndBackground = exponential.proliferation.neutral_mutation
60    {
61        // this is important: we update all background mutations at this
62        // time such that all cells are all on the same page.
63        // Since background mutations are implemented at each division, and
64        // cells do not proliferate at the same rate, we need to correct and
65        // update the background mutations at the timepoint corresponding
66        // to sampling step, i.e. before saving.
67        //
68        // Moreover there is delay between the end of the exp. growing phase
69        // and birth, hence we use `TIME_AT_BIRTH`. But this is true only for
70        // the Development phase.
71        debug!("updating the neutral background mutations for all cells");
72        // assign missing mutations
73        for stem_cell in exponential.subclones.get_mut_cells() {
74            if stem_cell.get_last_division_time() < &current_time {
75                assign_background_mutations(
76                    stem_cell,
77                    current_time,
78                    &exponential.distributions.neutral_poisson,
79                    rng,
80                );
81            }
82            // restarting the time within cells
83            stem_cell.set_last_division_time(time_to_restart).unwrap();
84        }
85    }
86    Moran {
87        subclones: exponential.subclones,
88        counter_divisions: exponential.counter_divisions,
89        // restart the time within the process
90        time: time_to_restart,
91        snapshots: process_options.snapshots,
92        path2dir: process_options.path,
93        filename,
94        distributions,
95        save_sfs_only,
96        save_population,
97        proliferation: exponential.proliferation,
98    }
99}

139        //! This is useful when we undersample the Moran process to a smaller

153impl AdvanceStep<MAX_SUBCLONES> for Exponential {
154    /// The id of a subclone [`CloneId`].
155    type Reaction = CloneId;
156
157    fn advance_step(&mut self, reaction: NextReaction<Self::Reaction>, rng: &mut impl Rng) {
158        //! Update the process by simulating the next proliferative event
159        //! according to the next `reaction` determined by the [Gillespie
160        //! algorithm](`sosa`).
161        //!
162        //! The proliferation step is implemented as following:
163        //!
164        //! 1. select the cell that will proliferate next from the clone
165        //!    with id `reaction` determined by the Gillespie algorithm
166        //!
167        //! 2. for the proliferating cell and its daughter cell:
168        //!     * mutate genome by storing a neutral number of mutations
169        //!
170        //!     * assign to new subclone with a probability determined by the
171        //!       rate of mutations conferring a proliferative advantage
172        // The Gillespie sampler samples the clone that will proliferate next,
173        // that is the clone with id `reaction.event`.
174        // Pick random proliferating cells from this clone. **Note that this
175        // removes the cells from the clone with id `reaction.event`.**
176        self.counter_divisions += 1;
177        self.time += reaction.time;
178        self.proliferation.proliferate(
179            &mut self.subclones,
180            self.time,
181            reaction.event,
182            &self.distributions,
183            rng,
184        );
185    }
186
187    fn update_state(&self, state: &mut CurrentState<MAX_SUBCLONES>) {
188        state.population = Variants::variant_counts(&self.subclones);
189    }
190}

260    fn keep_const_population_upon_symmetric_division(&mut self, rng: &mut impl Rng) {
261        //! If an symmetric division is performed, need to remove a random cell
262        //! from the population.
263        //! The cell having just proliferated can be removed as well, any cell
264        //! can be removed.
265        //!
266        //! We proceed as following:
267        //!     1. check if there is only one clone in the population, then
268        //!     remove a cell from this clone
269        //!     2. else, compute the variant counts
270        //!     3. and sample from any clone based on the weights defined by
271        //!     the variant counts
272        trace!("keeping the cell population constant");
273        // remove a cell from a random subclone based on the frequencies of
274        // the clones at the current state
275        let variants = Variants::variant_counts(&self.subclones);
276        let id2remove = WeightedIndex::new(variants).unwrap().sample(rng);
277        let _ = self
278            .subclones
279            .get_mut_clone_unchecked(id2remove)
280            .random_cell(rng)
281            .with_context(|| "found empty subclone")
282            .unwrap();
283        trace!("removing one cell from clone {id2remove}");
284    }

286    pub fn save(
287        &mut self,
288        time: f32,
289        saving_cells: &SavingCells,
290        save_sfs_only: bool,
291        rng: &mut impl Rng,
292    ) -> anyhow::Result<()> {
293        info!("saving process at time {time}");
294        if let NeutralMutations::UponDivisionAndBackground = self.proliferation.neutral_mutation {
295            // this is important: we update all background mutations at this
296            // time such that all cells are all on the same page.
297            // Since background mutations are implemented at each division, and
298            // cells do not proliferate at the same rate, we need to correct and
299            // update the background mutations at the timepoint corresponding
300            // to sampling step, i.e. before saving.
301            debug!("updating the neutral background mutations for all cells");
302            for stem_cell in self.subclones.get_mut_cells() {
303                assign_background_mutations(
304                    stem_cell,
305                    self.time,
306                    &self.distributions.neutral_poisson,
307                    rng,
308                );
309            }
310        }
311        let population = self.subclones.get_cells_with_clones_idx();
312        trace!("saving {saving_cells:#?}");
313        match saving_cells {
314            SavingCells::WholePopulation => save_it(
315                &self.path2dir,
316                &self.filename,
317                time,
318                population,
319                save_sfs_only,
320            )
321            .with_context(|| "cannot save the full population")
322            .unwrap(),
323            SavingCells::Subsampling {
324                sample_size,
325                population_as_well: populaiton_as_well,
326            } => {
327                if *populaiton_as_well {
328                    save_it(
329                        &self.path2dir,
330                        &self.filename,
331                        time,
332                        population,
333                        save_sfs_only,
334                    )
335                    .with_context(|| "cannot save the full population")
336                    .unwrap();
337                }
338                let cells_with_idx = self
339                    .subclones
340                    .get_cells_subsampled_with_clones_idx(*sample_size, rng);
341                assert_eq!(cells_with_idx.len(), *sample_size);
342                save_it(
343                    &self.path2dir,
344                    &self.filename,
345                    time,
346                    cells_with_idx,
347                    save_sfs_only,
348                )
349                .with_context(|| "cannot save the subsample")
350                .unwrap();
351            }
352        }
353        Ok(())
354    }

356    pub fn into_subsampled(self, nb_cells: NonZeroUsize, rng: &mut impl Rng) -> Self {
357        //! Take a subsample (without replacement) of the population of this
358        //! process using [`choose_multiple`](https://docs.rs/rand/latest/rand/seq/trait.IndexedRandom.html#method.choose_multiple).
359        let mut moran: Moran = self;
360        moran.subclones = moran.subclones.into_subsampled(nb_cells, rng);
361        debug!("Subsampled {} cells", moran.subclones.get_cells().len());
362
363        moran
364    }

367impl AdvanceStep<MAX_SUBCLONES> for Moran {
368    /// The id of a subclone [`CloneId`].
369    type Reaction = CloneId;
370
371    fn advance_step(&mut self, reaction: NextReaction<Self::Reaction>, rng: &mut impl Rng) {
372        //! Update the process by simulating the next proliferative event
373        //! according to the next `reaction` determined by the [Gillespie
374        //! algorithm](`sosa`).
375        //! Here we also save the state of the simulation at different
376        //! timepoints, see `self.snapshot`.
377        //!
378        //! For every simulated timestep, the algorithm samples an exponential
379        //! waiting time and the non-empty clone that will proliferate.
380        //! From the proliferating clone selected, we sample a random cell to
381        //! which we assign neutral and fit mutations.
382        // take snapshot
383        while !self.snapshots.is_empty() && self.snapshots.iter().any(|s| self.time >= s.time) {
384            let snapshot = self.snapshots.pop_front().unwrap();
385            debug!(
386                    "saving state for timepoint at simulation's time {} for timepoint at {:#?} with {} cells",
387                    self.time, snapshot.time, snapshot.cells2sample
388                );
389            let saving_cells =
390                if snapshot.cells2sample == self.subclones.compute_tot_cells() as usize {
391                    SavingCells::WholePopulation
392                } else {
393                    SavingCells::Subsampling {
394                        sample_size: snapshot.cells2sample,
395                        population_as_well: self.save_population,
396                    }
397                };
398            self.save(self.time, &saving_cells, self.save_sfs_only, rng)
399                .expect("cannot save snapshot");
400        }
401
402        // 1. select a cell `c` at random
403        // the SSA samples the clone that will proliferate next, that is the
404        // clone with id `reaction.event`. Pick random proliferating cells from
405        // this clone. **Note that this removes the cells from the clone with
406        // id `reaction.event`.**
407        self.time += reaction.time;
408        self.counter_divisions += 1;
409        self.proliferation.proliferate(
410            &mut self.subclones,
411            self.time,
412            reaction.event,
413            &self.distributions,
414            rng,
415        );
416
417        // remove a cell from the population
418        self.keep_const_population_upon_symmetric_division(rng);
419
420        trace!("{} cells", self.subclones.compute_tot_cells());
421    }
422
423    fn update_state(&self, state: &mut CurrentState<MAX_SUBCLONES>) {
424        state.population = Variants::variant_counts(&self.subclones);
425    }
426}

690    fn the_only_one_subclone_present(subclones: &SubClones) -> Option<CloneId> {
691        //! returns `None` if more than one subclone is present else the clone
692        //! id.
693        //!
694        //! **warning** this is very slow.
695        let tot_cells = subclones.compute_tot_cells();
696        (0..MAX_SUBCLONES).find(|&id| subclones.get_clone(id).unwrap().cell_count() == tot_cells)
697    }
