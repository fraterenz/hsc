69pub fn switch_to_moran(
70    mut exponential: Exponential,
71    process_options: ProcessOptions,
72    distributions: Distributions,
73    filename: PathBuf,
74    save_sfs_only: bool,
75    save_population: bool,
76    rng: &mut impl Rng,
77) -> Moran {
78    //! End the exponential growing phase and switch to a fixed-size
79    //! population phase.
80    //!
81    //! There is a delay between birth and the end of the exponential phase,
82    //! since stem cells stop dividing exponentially before birth.
83    //! We add background mutations in this interval of time between the
84    //! end of the exponentially growing phase and the Moran process.
85    if exponential.verbosity > 0 {
86        println!("switching to Moran at time {}", exponential.time);
87    }
88    let (current_time, time_to_restart) = match exponential.phase {
89        ExponentialPhase::Development => (TIME_AT_BIRTH, 0f32),
90        ExponentialPhase::Regrowth => (exponential.time, exponential.time),
91    };
92    if let NeutralMutations::UponDivisionAndBackground = exponential.proliferation.neutral_mutation
93    {
94        // this is important: we update all background mutations at this
95        // time such that all cells are all on the same page.
96        // Since background mutations are implemented at each division, and
97        // cells do not proliferate at the same rate, we need to correct and
98        // update the background mutations at the timepoint corresponding
99        // to sampling step, i.e. before saving.
100        //
101        // Moreover there is delay between the end of the exp. growing phase
102        // and birth, hence we use `TIME_AT_BIRTH`. But this is true only for
103        // the Development phase.
104        if exponential.verbosity > 0 {
105            println!("updating the neutral background mutations for all cells");
106        }
107        // assign missing mutations
108        for stem_cell in exponential.subclones.get_mut_cells() {
109            if stem_cell.get_last_division_time() < &current_time {
110                assign_background_mutations(
111                    stem_cell,
112                    current_time,
113                    &exponential.distributions.neutral_poisson,
114                    rng,
115                    exponential.verbosity,
116                );
117            }
118            // restarting the time within cells
119            stem_cell.set_last_division_time(time_to_restart).unwrap();
120        }
121    }
122    Moran {
123        subclones: exponential.subclones,
124        counter_divisions: exponential.counter_divisions,
125        // restart the time within the process
126        time: time_to_restart,
127        snapshots: process_options.snapshots,
128        path2dir: process_options.path,
129        verbosity: exponential.verbosity,
130        filename,
131        distributions,
132        save_sfs_only,
133        save_population,
134        proliferation: exponential.proliferation,
135    }
136}

196impl AdvanceStep<MAX_SUBCLONES> for Exponential {
197    /// The id of a subclone [`CloneId`].
198    type Reaction = CloneId;
199
200    fn advance_step(&mut self, reaction: NextReaction<Self::Reaction>, rng: &mut impl Rng) {
201        //! Update the process by simulating the next proliferative event
202        //! according to the next `reaction` determined by the [Gillespie
203        //! algorithm](`sosa`).
204        //!
205        //! The proliferation step is implemented as following:
206        //!
207        //! 1. select the cell that will proliferate next from the clone
208        //!    with id `reaction` determined by the Gillespie algorithm
209        //!
210        //! 2. for the proliferating cell and its daughter cell:
211        //!     * mutate genome by storing a neutral number of mutations
212        //!
213        //!     * assign to new subclone with a probability determined by the
214        //!       rate of mutations conferring a proliferative advantage
215        // The Gillespie sampler samples the clone that will proliferate next,
216        // that is the clone with id `reaction.event`.
217        // Pick random proliferating cells from this clone. **Note that this
218        // removes the cells from the clone with id `reaction.event`.**
219        self.counter_divisions += 1;
220        self.time += reaction.time;
221        self.proliferation.proliferate(
222            &mut self.subclones,
223            self.time,
224            reaction.event,
225            &self.distributions,
226            rng,
227            self.verbosity,
228        );
229    }
230
231    fn update_state(&self, state: &mut CurrentState<MAX_SUBCLONES>) {
232        state.population = Variants::variant_counts(&self.subclones);
233    }
234}

310    fn keep_const_population_upon_symmetric_division(&mut self, rng: &mut impl Rng) {
311        //! If an symmetric division is performed, need to remove a random cell
312        //! from the population.
313        //! The cell having just proliferated can be removed as well, any cell
314        //! can be removed.
315        //!
316        //! We proceed as following:
317        //!     1. check if there is only one clone in the population, then
318        //!     remove a cell from this clone
319        //!     2. else, compute the variant counts
320        //!     3. and sample from any clone based on the weights defined by
321        //!     the variant counts
322        if self.verbosity > 2 {
323            println!("keeping the cell population constant");
324        }
325        // remove a cell from a random subclone based on the frequencies of
326        // the clones at the current state
327        let variants = Variants::variant_counts(&self.subclones);
328        let id2remove = WeightedIndex::new(variants).unwrap().sample(rng);
329        let _ = self
330            .subclones
331            .get_mut_clone_unchecked(id2remove)
332            .random_cell(rng)
333            .with_context(|| "found empty subclone")
334            .unwrap();
335        if self.verbosity > 2 {
336            println!("removing one cell from clone {id2remove}");
337        }
338    }

362    fn save_it(
363        &self,
364        time: f32,
365        cells_with_idx: Vec<(&StemCell, usize)>,
366        save_sfs_only: bool,
367    ) -> anyhow::Result<()> {
368        if self.verbosity > 0 {
369            println!("saving data at time {time}");
370        }
371        let cells: Vec<&StemCell> = cells_with_idx.iter().map(|ele| ele.0).collect();
372        let nb_cells = cells.len();
373
374        if self.verbosity > 0 {
375            println!("saving {nb_cells} cells");
376        }
377
378        Sfs::from_cells(&cells, self.verbosity)
379            .unwrap_or_else(|_| panic!("cannot create SFS for timepoint at time {time}"))
380            .save(
381                &self.make_path(Stats2Save::Sfs, nb_cells, time)?,
382                self.verbosity,
383            )?;
384
385        if !save_sfs_only {
386            MutationalBurden::from_cells(&cells, self.verbosity)
387                .unwrap_or_else(|_| panic!("cannot create burden for the timepoint at time {time}"))
388                .save(
389                    &self.make_path(Stats2Save::Burden, nb_cells, time)?,
390                    self.verbosity,
391                )?;
392            save_variant_fraction(
393                &SubClones::from(
394                    cells_with_idx
395                        .into_iter()
396                        .map(|(cell, id)| (cell.to_owned(), id))
397                        .collect::<Vec<(StemCell, usize)>>(),
398                ),
399                &self.make_path(Stats2Save::VariantFraction, nb_cells, time)?,
400                self.verbosity,
401            )?;
402        }
403
404        if self.verbosity > 0 {
405            println!(
406                "saved measurements after {} mutational events",
407                self.counter_divisions
408            );
409        }
410        Ok(())
411    }

413    pub fn save(
414        &mut self,
415        time: f32,
416        saving_cells: &SavingCells,
417        save_sfs_only: bool,
418        rng: &mut impl Rng,
419    ) -> anyhow::Result<()> {
420        if self.verbosity > 0 {
421            println!("saving process at time {time}");
422        }
423        if let NeutralMutations::UponDivisionAndBackground = self.proliferation.neutral_mutation {
424            // this is important: we update all background mutations at this
425            // time such that all cells are all on the same page.
426            // Since background mutations are implemented at each division, and
427            // cells do not proliferate at the same rate, we need to correct and
428            // update the background mutations at the timepoint corresponding
429            // to sampling step, i.e. before saving.
430            if self.verbosity > 0 {
431                println!("updating the neutral background mutations for all cells");
432            }
433            for stem_cell in self.subclones.get_mut_cells() {
434                assign_background_mutations(
435                    stem_cell,
436                    self.time,
437                    &self.distributions.neutral_poisson,
438                    rng,
439                    self.verbosity,
440                );
441            }
442        }
443        let population = self.subclones.get_cells_with_clones_idx();
444        if self.verbosity > 0 {
445            println!("saving {saving_cells:#?}");
446        }
447        match saving_cells {
448            SavingCells::WholePopulation => self
449                .save_it(time, population, save_sfs_only)
450                .with_context(|| "cannot save the full population")
451                .unwrap(),
452            SavingCells::Subsampling {
453                sample_size,
454                population_as_well: populaiton_as_well,
455            } => {
456                if *populaiton_as_well {
457                    self.save_it(time, population, save_sfs_only)
458                        .with_context(|| "cannot save the full population")
459                        .unwrap();
460                }
461                let cells_with_idx = self
462                    .subclones
463                    .get_cells_subsampled_with_clones_idx(*sample_size, rng);
464                assert_eq!(cells_with_idx.len(), *sample_size);
465                self.save_it(time, cells_with_idx, save_sfs_only)
466                    .with_context(|| "cannot save the subsample")
467                    .unwrap();
468            }
469        }
470        Ok(())
471    }

486impl AdvanceStep<MAX_SUBCLONES> for Moran {
487    /// The id of a subclone [`CloneId`].
488    type Reaction = CloneId;
489
490    fn advance_step(&mut self, reaction: NextReaction<Self::Reaction>, rng: &mut impl Rng) {
491        //! Update the process by simulating the next proliferative event
492        //! according to the next `reaction` determined by the [Gillespie
493        //! algorithm](`sosa`).
494        //! Here we also save the state of the simulation at different
495        //! timepoints, see `self.snapshot`.
496        //!
497        //! For every simulated timestep, the algorithm samples an exponential
498        //! waiting time and the non-empty clone that will proliferate.
499        //! From the proliferating clone selected, we sample a random cell to
500        //! which we assign neutral and fit mutations.
501        // take snapshot
502        while !self.snapshots.is_empty() && self.snapshots.iter().any(|s| self.time >= s.time) {
503            let snapshot = self.snapshots.pop_front().unwrap();
504            if self.verbosity > 0 {
505                println!(
506                    "saving state for timepoint at simulation's time {} for timepoint at {:#?} with {} cells",
507                    self.time, snapshot.time, snapshot.cells2sample
508                );
509            }
510            let saving_cells =
511                if snapshot.cells2sample == self.subclones.compute_tot_cells() as usize {
512                    SavingCells::WholePopulation
513                } else {
514                    SavingCells::Subsampling {
515                        sample_size: snapshot.cells2sample,
516                        population_as_well: self.save_population,
517                    }
518                };
519            self.save(self.time, &saving_cells, self.save_sfs_only, rng)
520                .expect("cannot save snapshot");
521        }
522
523        // 1. select a cell `c` at random
524        // the SSA samples the clone that will proliferate next, that is the
525        // clone with id `reaction.event`. Pick random proliferating cells from
526        // this clone. **Note that this removes the cells from the clone with
527        // id `reaction.event`.**
528        self.time += reaction.time;
529        self.counter_divisions += 1;
530        self.proliferation.proliferate(
531            &mut self.subclones,
532            self.time,
533            reaction.event,
534            &self.distributions,
535            rng,
536            self.verbosity,
537        );
538
539        // remove a cell from the population
540        self.keep_const_population_upon_symmetric_division(rng);
541
542        if self.verbosity > 2 {
543            println!("{} cells", self.subclones.compute_tot_cells());
544        }
545    }

547    fn update_state(&self, state: &mut CurrentState<MAX_SUBCLONES>) {
548        state.population = Variants::variant_counts(&self.subclones);
549    }

694    fn create_exp(fit_variants: bool, cells: NonZeroU64, is_regrowht: bool) -> Exponential {
695        let mu = if fit_variants { 0.999 } else { 0. };
696        let phase = if is_regrowht {
697            ExponentialPhase::Regrowth
698        } else {
699            ExponentialPhase::Development
700        };
701        Exponential::new(
702            SubClones::new(vec![StemCell::new(); cells.get() as usize], 3, 0),
703            Distributions::new(Probs::new(10., 1., mu, 1., cells.get(), 0), 0),
704            Proliferation::default(),
705            0.,
706            phase,
707            0,
708        )
709    }

816    fn the_only_one_subclone_present(subclones: &SubClones) -> Option<CloneId> {
817        //! returns `None` if more than one subclone is present else the clone
818        //! id.
819        //!
820        //! **warning** this is very slow.
821        let tot_cells = subclones.compute_tot_cells();
822        (0..MAX_SUBCLONES).find(|&id| subclones.get_clone(id).unwrap().cell_count() == tot_cells)
823    }
