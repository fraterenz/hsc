1//! The stochastic simulation algorithm ([SSA](https://en.wikipedia.org/wiki/Gillespie_algorithm#Algorithm))
2//! with a Monte-Carlo generating method.
3//!
4//! ## Example
5//! `sosa` allows using the SSA with agents that carry some individual
6//! proprieties evolving over time.
7//!
8//! Consider for example human cells which reproduce asexually and are thought
9//! to acquire new point mutations in their genome upon cell division.
10//! We could be interested in tracking the evolution in the number of mutations
11//! over time, as cells reproduce.
12//! Moreover, cells can reproduce at different rates on average, e.g. cells
13//! carrying special mutations can reproduce faster compared to other cells.
14//! In this case, we can use `sosa` to perform SSA and at the same time track
15//! those mutations over time taking into account the different proliferation
16//! rates.
17//!
18//! Note that if we are just interested in tracking the number of individuals
19//! over time, without taking into consideration the indiviual proprities of
20//! the agents, then [`rebop`](https://crates.io/crates/rebop) should be used
21//! instead of `sosa`.

37/// The next reaction sampled by the SSA.
38#[derive(Debug)]
39pub struct NextReaction<Reaction>
40where
41    Reaction: std::fmt::Debug,
42{
43    /// The relative time at which this next reaction took place.
44    pub time: f32,
45    /// The event corresponding to the next iteration found by the
46    /// [`AdvanceStep::next_reaction`].
47    pub event: Reaction,
48}

50/// Whether to stop or continue the simulation.
51#[derive(Debug, Copy, Clone, PartialEq, Eq)]
52pub enum SimState {
53    /// A simulation is stopped whether one of those conditions are met (see
54    /// [`StopReason`]):
55    ///
56    /// 1. the maximal number of individuals has been reached,
57    /// 2. the total population size is 0 (all lineages have died out),
58    /// 3. the maximal number of iterations has been reached.
59    ///
60    Stop(StopReason),
61    Continue,
62}
63
64#[derive(Debug, Copy, Clone, Eq, PartialEq)]
65pub enum StopReason {
66    /// No individual left, all lineages have died out.
67    NoIndividualsLeft,
68    /// The maximal number of individual has been reached.
69    MaxIndividualsReached,
70    /// The maximal number of iterations has been reached.
71    MaxItersReached,
72    /// Absorbing state has been reached
73    AbsorbingStateReached,
74    /// The maximal timestep has been reached
75    MaxTimeReached,
76}

92/// The main loop running one realisation of a stochastic process with
93/// `NB_REACTIONS` possible `REACTION`s.
94pub fn simulate<P, REACTION, const NB_REACTIONS: usize>(
95    state: &mut CurrentState<NB_REACTIONS>,
96    rates: &ReactionRates<{ NB_REACTIONS }>,
97    possible_reactions: &[REACTION; NB_REACTIONS],
98    bd_process: &mut P,
99    options: &Options,
100    rng: &mut impl Rng,
101) -> StopReason
102where
103    P: AdvanceStep<NB_REACTIONS, Reaction = REACTION>,
104    REACTION: std::fmt::Debug,
105{
106    let mut iter = options.init_iter;
107    let mut time = 0f32;
108    loop {
109        if state.population.iter().sum::<u64>() >= options.max_cells {
110            return StopReason::MaxIndividualsReached;
111        }
112
113        let (sim_state, reaction) = bd_process.next_reaction(
114            state,
115            rates,
116            possible_reactions,
117            IterTime {
118                iter: options.max_iter_time.iter - 1,
119                time: options.max_iter_time.time,
120            },
121            IterTime { iter, time },
122            rng,
123        );
124
125        if options.verbosity > 1 {
126            println!("time: {time} iter: {iter} and reaction: {reaction:#?}");
127
128            if options.verbosity > 2 {
129                println!("State: {state:#?}");
130            }
131        }
132
133        match sim_state {
134            SimState::Continue => {
135                // unwrap is safe since SimState::Continue returns always
136                // something (i.e. not None).
137                let reaction = reaction.unwrap();
138                let reaction_time = reaction.time;
139
140                // update the process according to the reaction
141                bd_process.advance_step(reaction, rng);
142
143                // update the state according to the process
144                bd_process.update_state(state);
145                iter += 1;
146                time += reaction_time;
147            }
148            SimState::Stop(reason) => return reason,
149        }
150    }
151}

153/// The current state of a Markov process.
154#[derive(Debug, Clone)]
155pub struct CurrentState<const NB_REACTIONS: usize> {
156    /// The number of individuals for all reactions.
157    pub population: [NbIndividuals; NB_REACTIONS],
158}
159
160/// Perform an iteration of the SSA.
161pub trait AdvanceStep<const NB_REACTIONS: usize> {
162    type Reaction: std::fmt::Debug + Copy;
163
164    fn next_reaction(
165        &self,
166        state: &CurrentState<NB_REACTIONS>,
167        rates: &ReactionRates<{ NB_REACTIONS }>,
168        possible_reactions: &[Self::Reaction; NB_REACTIONS],
169        max_iter_and_time: IterTime,
170        iter_and_time: IterTime,
171        rng: &mut impl Rng,
172    ) -> (SimState, Option<NextReaction<Self::Reaction>>) {
173        //! Find the next reaction in the system according to a
174        //! [Monte-Carlo generating method](https://en.wikipedia.org/wiki/Gillespie_algorithm#Algorithm).
175        //! ## Returns
176        //! `None` if the maximal number of iterations have been reached or
177        //! there aren't any individuals left in the total population, see
178        //! [`SimState`].
179        //!
180        //! ## Panics
181        //! If no inidividuals left or all computed times are not normal.
182        // StopIteration appears when there are no cells anymore (due to
183        // cell death), when the iteration has reached the max number of
184        // iterations nb_iter >= self.max_iter or maximal number of cells
185        // i.e. when the iteration has generated a tumor of max_cells size
186        if state.population.iter().sum::<u64>() == 0u64 {
187            return (SimState::Stop(StopReason::NoIndividualsLeft), None);
188        };
189        if iter_and_time.iter >= max_iter_and_time.iter {
190            return (SimState::Stop(StopReason::MaxItersReached), None);
191        };
192        if iter_and_time.time >= max_iter_and_time.time {
193            return (SimState::Stop(StopReason::MaxTimeReached), None);
194        };
195
196        let mut selected_event = 0_usize;
197        let times = rates.compute_times_events(&state.population, rng).unwrap();
198        let mut smaller_waiting_time = times[0];
199        for (idx, &waiting_time) in times.iter().enumerate() {
200            if waiting_time <= smaller_waiting_time {
201                smaller_waiting_time = waiting_time;
202                selected_event = idx;
203            }
204        }
205        (
206            SimState::Continue,
207            Some(NextReaction {
208                time: smaller_waiting_time,
209                event: possible_reactions[selected_event],
210            }),
211        )
212    }
213
214    /// Updates the process by stepping one step forward in the sumulation
215    /// according to the [`NextReaction`] generated by
216    /// [`AdvanceStep::next_reaction`].
217    fn advance_step(&mut self, reaction: NextReaction<Self::Reaction>, rng: &mut impl Rng);
218
219    /// Once the process has been updated by [`AdvanceStep::advance_step`],
220    /// update back the [`CurrentState`] according to the process' rules.
221    fn update_state(&self, state: &mut CurrentState<NB_REACTIONS>);
222}

224/// Requesting to compute the reaction rates via [`ReactionRates`] when the
225/// simulation's state is not compatible (e.g. `NoIndividualsLeft`) raises an
226/// error.
227#[derive(Error, Debug, PartialEq)]
228enum ReactionRatesError {
229    #[error("no individuals left")]
230    NoIndividualsLeft,
231    #[error("the computed times are 0 or inf, probably 0 or inf rates")]
232    ComputeTimesAllInfinite,
233}
234
235/// The rate of a reaction is the average number of occurrence of that reaction
236/// in a time-unit.
237#[derive(Debug, Clone)]
238pub struct ReactionRates<const N: usize>(
239    /// All the `N` rates for the `N` reactions present in the system.
240    pub [f32; N],
241);
242
243impl<const N: usize> ReactionRates<N> {
244    fn compute_times_events(
245        &self,
246        population: &[NbIndividuals; N],
247        rng: &mut impl Rng,
248    ) -> Result<[f32; N], ReactionRatesError> {
249        //! Compute the Gillepsie-time for all reactions.
250        //! The Gillespie-time is
251        //! [defined](https://en.wikipedia.org/wiki/Gillespie_algorithm#Algorithm)
252        //! as:
253        //!
254        //! `-ln(1 - r) / (population[i] * rates[i]) for i 0..N`
255        //!
256        //! where `r` is a random number and rates is `self.0`.
257        //! ## Returns
258        //! - error when all computed times are infinity or 0
259        //! - array of computed times otherwise
260        if population.iter().all(|&pop| pop == 0) {
261            return Err(ReactionRatesError::NoIndividualsLeft);
262        }
263        let mut times = self.0;
264
265        for i in 0..N {
266            times[i] = exprand(times[i] * population[i] as f32, rng);
267        }
268        if times.iter().any(|time| time.is_normal()) {
269            return Ok(times);
270        }
271        Err(ReactionRatesError::ComputeTimesAllInfinite)
272    }
273}

275pub fn exprand(lambda: f32, rng: &mut impl Rng) -> f32 {
276    //! Generates a random waiting time using the exponential waiting time with
277    //! parameter `lambda` of Poisson StochasticProcess.
278    //!
279    //! ## Returns
280    //! - a waiting time of `0` if `lambda` is infinity,
281    //! - a random exponential waiting time if `lambda` [`f32::is_normal`],
282    //! - infinity otherwise.
283    //! ```
284    //! use rand::{rngs::SmallRng, SeedableRng};
285    //! # use sosa::exprand;
286    //!
287    //! let mut rng = SmallRng::seed_from_u64(1u64);
288    //!
289    //! let lambda_gr_than_zero = 0.1_f32;
290    //! assert!(exprand(lambda_gr_than_zero, &mut rng).is_sign_positive());
291    //!
292    //! let lambda_zero = 0_f32;
293    //! assert!(exprand(lambda_zero, &mut rng).is_infinite());
294    //!
295    //! let lambda_inf = f32::INFINITY;
296    //! assert!((exprand(lambda_inf, &mut rng) - 0.).abs() < f32::EPSILON);
297    //! ```
298    //!
299    //! ## Panics
300    //! When `lambda` is negative.
301    //!
302    //! ```should_panic
303    //! use rand::{rngs::SmallRng, SeedableRng};
304    //! # use sosa::exprand;
305    //!
306    //! let mut rng = SmallRng::seed_from_u64(1u64);
307    //!
308    //! let lambda_neg = -0.1_f32;
309    //! exprand(lambda_neg, &mut rng);
310    //! ```
311    assert!(!lambda.is_sign_negative());
312    if lambda.is_normal() {
313        // random number between (0, 1)
314        let val: f32 = rng.sample(Open01);
315        return -val.ln() / lambda;
316    } else if lambda.is_infinite() {
317        return 0.;
318    }
319    f32::INFINITY
320}

322pub fn write2file<T: std::fmt::Display>(
323    data: &[T],
324    path: &Path,
325    header: Option<&str>,
326    endline: bool,
327) -> anyhow::Result<()> {
328    //! Write vector of float into new file with a precision of 4 decimals.
329    //! Write NAN if the slice to write to file is empty.
330    fs::create_dir_all(path.parent().unwrap()).expect("Cannot create dir");
331    let f = fs::OpenOptions::new()
332        .read(true)
333        .append(true)
334        .create(true)
335        .open(path)
336        .with_context(|| "Cannot open stream")?;
337
338    let mut buffer = BufWriter::new(f);
339
340    if !data.is_empty() {
341        if let Some(h) = header {
342            writeln!(buffer, "{h}")?;
343        }
344
345        for ele in data.iter() {
346            write!(buffer, "{ele:.4},")?;
347        }
348
349        if endline {
350            writeln!(buffer)?;
351        }
352    } else {
353        write!(buffer, "{},", f32::NAN)?;
354    }
355
356    Ok(())
357}

367    fn exprand_same_seed_test(lambda: u8, seed: u64) -> bool {
368        let mut rng = SmallRng::seed_from_u64(seed);
369        let lambda = lambda as f32 / 10.;
370        if lambda.is_normal() {
371            let exp1 = exprand(lambda, &mut rng);
372            let mut rng = SmallRng::seed_from_u64(seed);
373            let exp2 = exprand(lambda, &mut rng);
374            return (exp1 - exp2).abs() < f32::EPSILON;
375        }

377    }
