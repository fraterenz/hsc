1//! The stochastic simulation algorithm ([SSA](https://en.wikipedia.org/wiki/Gillespie_algorithm#Algorithm))
2//! with a Monte-Carlo generating method.
3//!
4//! ## Example
5//! `sosa` allows using the SSA with agents that carry some individual
6//! proprieties evolving over time.
7//!
8//! Consider for example human cells which reproduce asexually and are thought
9//! to acquire new point mutations in their genome upon cell division.
10//! We could be interested in tracking the evolution in the number of mutations
11//! over time, as cells reproduce.
12//! Moreover, cells can reproduce at different rates on average, e.g. cells
13//! carrying special mutations can reproduce faster compared to other cells.
14//! In this case, we can use `sosa` to perform SSA and at the same time track
15//! those mutations over time taking into account the different proliferation
16//! rates.
17//!
18//! Note that if we are just interested in tracking the number of individuals
19//! over time, without taking into consideration the indiviual proprities of
20//! the agents, then [`rebop`](https://crates.io/crates/rebop) should be used
21//! instead of `sosa`.

39pub struct NextReaction<Reaction>
40where
41    Reaction: std::fmt::Debug,
42{
43    /// The relative time at which this next reaction took place.

47    pub event: Reaction,

50/// Whether to stop or continue the simulation.
51#[derive(Debug, Copy, Clone, PartialEq, Eq)]
52pub enum SimState {
53    /// A simulation is stopped whether one of those conditions are met (see
54    /// [`StopReason`]):
55    ///
56    /// 1. the maximal number of individuals has been reached,
57    /// 2. the total population size is 0 (all lineages have died out),
58    /// 3. the maximal number of iterations has been reached.
59    ///
60    Stop(StopReason),
61    Continue,
62}
63
64#[derive(Debug, Copy, Clone, Eq, PartialEq)]
65pub enum StopReason {
66    /// No individual left, all lineages have died out.
67    NoIndividualsLeft,
68    /// The maximal number of individual has been reached.
69    MaxIndividualsReached,
70    /// The maximal number of iterations has been reached.
71    MaxItersReached,
72    /// Absorbing state has been reached
73    AbsorbingStateReached,
74    /// The maximal timestep has been reached
75    MaxTimeReached,
76}

92/// The main loop running one realisation of a stochastic process with
93/// `NB_REACTIONS` possible `REACTION`s.
94/// ### Example
95/// Here we todo.
96///
97/// ```rust
98/// # use rand::SeedableRng;
99/// # use rand::Rng;
100/// # use rand::rngs::SmallRng;
101/// # use sosa::CurrentState;
102/// # use sosa::AdvanceStep;
103/// # use sosa::NextReaction;
104/// struct HealthyCells {
105///     // a population of 3 types of cells
106///     population: [u64; 3],
107/// }
108/// impl AdvanceStep<2> for HealthyCells {
109///     type Reaction = usize;
110///     fn advance_step(&mut self, reaction: NextReaction<Self::Reaction>, _rng: &mut impl Rng) {
111///         // a new cell is born!
112///         self.population[reaction.event] += 1;
113///     }
114///     fn update_state(&self, state: &mut CurrentState<2>) {
115///         // provide some flexibility: `state` and the `population` don't need to be
116///         // the same. Here for example the state has only 2 types
117///         state.population = [self.population[0], self.population[1] + self.population[2]];
118///     }
119/// }
120///
121/// # let seed = 26;
122/// # let mut rng = SmallRng::seed_from_u64(seed);
123/// let population = [0, 0, 0];
124/// let mut process = HealthyCells { population };
125/// // type of the cell that will proliferate
126/// let type_proliferating = 1;
127/// let mut state = CurrentState { population: [population[0], population[1]] };
128/// let next_reaction = NextReaction {event: type_proliferating, time: 0.2};
129/// let mut expected_population = population;
130/// expected_population[type_proliferating] = population[type_proliferating] + 1;
131///
132/// // a new cell of type 1 is born
133/// process.advance_step(next_reaction, &mut rng);
134/// assert_eq!(process.population, expected_population);
135/// // but the state hasnt changed yet
136/// assert_eq!(state.population, population[..2]);
137///
138/// // update the state now
139/// process.update_state(&mut state);
140/// assert_eq!(state.population, expected_population[..2]);
141/// ```
142pub fn simulate<P, REACTION, const NB_REACTIONS: usize>(
143    state: &mut CurrentState<NB_REACTIONS>,
144    rates: &ReactionRates<{ NB_REACTIONS }>,
145    possible_reactions: &[REACTION; NB_REACTIONS],
146    bd_process: &mut P,
147    options: &Options,
148    rng: &mut impl Rng,
149) -> StopReason
150where
151    P: AdvanceStep<NB_REACTIONS, Reaction = REACTION>,
152    REACTION: std::fmt::Debug,
153{
154    let mut iter = options.init_iter;
155    let mut time = 0f32;
156    loop {
157        if state.population.iter().sum::<u64>() >= options.max_cells {
158            return StopReason::MaxIndividualsReached;
159        }
160
161        let (sim_state, reaction) = bd_process.next_reaction(
162            state,
163            rates,
164            possible_reactions,
165            IterTime {
166                iter: options.max_iter_time.iter - 1,
167                time: options.max_iter_time.time,
168            },
169            IterTime { iter, time },
170            rng,
171        );
172
173        if options.verbosity > 1 {
174            println!("time: {time} iter: {iter} and reaction: {reaction:#?}");
175
176            if options.verbosity > 2 {
177                println!("State: {state:#?}");
178            }
179        }
180
181        match sim_state {
182            SimState::Continue => {
183                // unwrap is safe since SimState::Continue returns always
184                // something (i.e. not None).
185                let reaction = reaction.unwrap();
186                let reaction_time = reaction.time;
187
188                // update the process according to the reaction
189                bd_process.advance_step(reaction, rng);
190
191                // update the state according to the process
192                bd_process.update_state(state);
193                iter += 1;
194                time += reaction_time;
195            }
196            SimState::Stop(reason) => return reason,
197        }
198    }
199}

201/// The current state of a Markov process.
202#[derive(Debug, Clone)]
203pub struct CurrentState<const NB_REACTIONS: usize> {
204    /// The number of individuals for all reactions.
205    pub population: [NbIndividuals; NB_REACTIONS],
206}
207
208/// Perform an iteration of the SSA.
209pub trait AdvanceStep<const NB_REACTIONS: usize> {
210    type Reaction: std::fmt::Debug + Copy;
211
212    fn next_reaction(
213        &self,
214        state: &CurrentState<NB_REACTIONS>,
215        rates: &ReactionRates<{ NB_REACTIONS }>,
216        possible_reactions: &[Self::Reaction; NB_REACTIONS],
217        max_iter_and_time: IterTime,
218        iter_and_time: IterTime,
219        rng: &mut impl Rng,
220    ) -> (SimState, Option<NextReaction<Self::Reaction>>) {
221        //! Find the next reaction in the system according to a
222        //! [Monte-Carlo generating method](https://en.wikipedia.org/wiki/Gillespie_algorithm#Algorithm).
223        //! ## Returns
224        //! `None` if the maximal number of iterations have been reached or
225        //! there aren't any individuals left in the total population, see
226        //! [`SimState`].
227        //!
228        //! ## Panics
229        //! If no inidividuals left or all computed times are not normal.
230        // StopIteration appears when there are no cells anymore (due to
231        // cell death), when the iteration has reached the max number of
232        // iterations nb_iter >= self.max_iter or maximal number of cells
233        // i.e. when the iteration has generated a tumor of max_cells size
234        if state.population.iter().sum::<u64>() == 0u64 {
235            return (SimState::Stop(StopReason::NoIndividualsLeft), None);
236        };
237        if iter_and_time.iter >= max_iter_and_time.iter {
238            return (SimState::Stop(StopReason::MaxItersReached), None);
239        };
240        if iter_and_time.time >= max_iter_and_time.time {
241            return (SimState::Stop(StopReason::MaxTimeReached), None);
242        };
243
244        let mut selected_event = 0_usize;
245        let times = rates.compute_times_events(&state.population, rng).unwrap();
246        let mut smaller_waiting_time = times[0];
247        for (idx, &waiting_time) in times.iter().enumerate() {
248            if waiting_time <= smaller_waiting_time {
249                smaller_waiting_time = waiting_time;
250                selected_event = idx;
251            }
252        }
253        (
254            SimState::Continue,
255            Some(NextReaction {
256                time: smaller_waiting_time,
257                event: possible_reactions[selected_event],
258            }),
259        )
260    }
261
262    /// Updates the process by stepping one step forward in the sumulation
263    /// according to the [`NextReaction`] generated by
264    /// [`AdvanceStep::next_reaction`].
265    fn advance_step(&mut self, reaction: NextReaction<Self::Reaction>, rng: &mut impl Rng);
266
267    /// Once the process has been updated by [`AdvanceStep::advance_step`],
268    /// update back the [`CurrentState`] according to the process' rules.
269    fn update_state(&self, state: &mut CurrentState<NB_REACTIONS>);
270}

272/// Requesting to compute the reaction rates via [`ReactionRates`] when the
273/// simulation's state is not compatible (e.g. `NoIndividualsLeft`) raises an
274/// error.
275#[derive(Error, Debug, PartialEq)]
276enum ReactionRatesError {
277    #[error("no individuals left")]
278    NoIndividualsLeft,
279    #[error("the computed times are 0 or inf, probably 0 or inf rates")]
280    ComputeTimesAllInfinite,
281}

283/// The rate of a reaction is the average number of occurrence of that reaction
284/// in a time-unit.
285#[derive(Debug, Clone)]
286pub struct ReactionRates<const N: usize>(
287    /// All the `N` rates for the `N` reactions present in the system.
288    pub [f32; N],
289);
290
291impl<const N: usize> ReactionRates<N> {
292    fn compute_times_events(
293        &self,
294        population: &[NbIndividuals; N],
295        rng: &mut impl Rng,
296    ) -> Result<[f32; N], ReactionRatesError> {
297        //! Compute the Gillepsie-time for all reactions.
298        //! The Gillespie-time is
299        //! [defined](https://en.wikipedia.org/wiki/Gillespie_algorithm#Algorithm)
300        //! as:
301        //!
302        //! `-ln(1 - r) / (population[i] * rates[i]) for i 0..N`
303        //!
304        //! where `r` is a random number and rates is `self.0`.
305        //! ## Returns
306        //! - error when all computed times are infinity or 0
307        //! - array of computed times otherwise
308        if population.iter().all(|&pop| pop == 0) {
309            return Err(ReactionRatesError::NoIndividualsLeft);
310        }
311        let mut times = self.0;
312
313        for i in 0..N {
314            times[i] = exprand(times[i] * population[i] as f32, rng);
315        }
316        if times.iter().any(|time| time.is_normal()) {
317            return Ok(times);
318        }
319        Err(ReactionRatesError::ComputeTimesAllInfinite)
320    }
321}

323pub fn exprand(lambda: f32, rng: &mut impl Rng) -> f32 {
324    //! Generates a random waiting time using the exponential waiting time with
325    //! parameter `lambda` of Poisson StochasticProcess.
326    //!
327    //! ## Returns
328    //! - a waiting time of `0` if `lambda` is infinity,
329    //! - a random exponential waiting time if `lambda` [`f32::is_normal`],
330    //! - infinity otherwise.
331    //! ```
332    //! use rand::{rngs::SmallRng, SeedableRng};
333    //! # use sosa::exprand;
334    //!
335    //! let mut rng = SmallRng::seed_from_u64(1u64);
336    //!
337    //! let lambda_gr_than_zero = 0.1_f32;
338    //! assert!(exprand(lambda_gr_than_zero, &mut rng).is_sign_positive());
339    //!
340    //! let lambda_zero = 0_f32;
341    //! assert!(exprand(lambda_zero, &mut rng).is_infinite());
342    //!
343    //! let lambda_inf = f32::INFINITY;
344    //! assert!((exprand(lambda_inf, &mut rng) - 0.).abs() < f32::EPSILON);
345    //! ```
346    //!
347    //! ## Panics
348    //! When `lambda` is negative.
349    //!
350    //! ```should_panic
351    //! use rand::{rngs::SmallRng, SeedableRng};
352    //! # use sosa::exprand;
353    //!
354    //! let mut rng = SmallRng::seed_from_u64(1u64);
355    //!
356    //! let lambda_neg = -0.1_f32;
357    //! exprand(lambda_neg, &mut rng);
358    //! ```
359    assert!(!lambda.is_sign_negative());
360    if lambda.is_normal() {
361        // random number between (0, 1)
362        let val: f32 = rng.sample(Open01);
363        return -val.ln() / lambda;
364    } else if lambda.is_infinite() {
365        return 0.;
366    }
367    f32::INFINITY
368}

370pub fn write2file<T: std::fmt::Display>(
371    data: &[T],
372    path: &Path,
373    header: Option<&str>,
374    endline: bool,
375) -> anyhow::Result<()> {
376    //! Write vector of float into new file with a precision of 4 decimals.
377    //! Write NAN if the slice to write to file is empty.
378    fs::create_dir_all(path.parent().unwrap()).expect("Cannot create dir");
379    let f = fs::OpenOptions::new()
380        .read(true)
381        .append(true)
382        .create(true)
383        .open(path)
384        .with_context(|| "Cannot open stream")?;
385
386    let mut buffer = BufWriter::new(f);
387
388    if !data.is_empty() {
389        if let Some(h) = header {
390            writeln!(buffer, "{h}")?;
391        }
392
393        for ele in data.iter() {
394            write!(buffer, "{ele:.4},")?;
395        }
396
397        if endline {
398            writeln!(buffer)?;
399        }
400    } else {
401        write!(buffer, "{},", f32::NAN)?;
402    }
403
404    Ok(())
405}

415    fn exprand_same_seed_test(lambda: u8, seed: u64) -> bool {
416        let mut rng = SmallRng::seed_from_u64(seed);
417        let lambda = lambda as f32 / 10.;
418        if lambda.is_normal() {
419            let exp1 = exprand(lambda, &mut rng);
420            let mut rng = SmallRng::seed_from_u64(seed);
421            let exp2 = exprand(lambda, &mut rng);
422            return (exp1 - exp2).abs() < f32::EPSILON;
423        }
424        exprand(lambda, &mut rng).is_infinite()
425    }
