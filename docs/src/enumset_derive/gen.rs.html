1//! This module handles generating the actual code to allow an enum type to be used as a bitset.
2
3use crate::plan::{EnumSetInfo, InternalRepr, SerdeRepr};
4use proc_macro2::{Literal, Span, TokenStream as SynTokenStream};
5use quote::*;
6use syn::{Lit, LitInt};
7
8struct Paths {
9    enumset: SynTokenStream,
10    typed_enumset: SynTokenStream,
11    core: SynTokenStream,
12    internal: SynTokenStream,
13    serde: SynTokenStream,
14}
15impl Paths {
16    fn from(info: &EnumSetInfo) -> Paths {
17        let name = &info.name;
18        let enumset = match &info.crate_name {
19            Some(crate_name) => quote!(::#crate_name),
20            None => {
21                #[cfg(feature = "proc-macro-crate")]
22                {
23                    use proc_macro_crate::FoundCrate;
24
25                    let crate_name = proc_macro_crate::crate_name("enumset");
26                    match crate_name {
27                        Ok(FoundCrate::Name(name)) => {
28                            let ident =
29                                proc_macro2::Ident::new(&name, proc_macro2::Span::call_site());
30                            quote!(::#ident)
31                        }
32                        _ => quote!(::enumset),
33                    }
34                }
35
36                #[cfg(not(feature = "proc-macro-crate"))]
37                {
38                    quote!(::enumset)
39                }
40            }
41        };
42        let typed_enumset = quote!(#enumset::EnumSet<#name>);
43        let core = quote!(#enumset::__internal::core_export);
44        let internal = quote!(#enumset::__internal);
45        let serde = quote!(#enumset::__internal::serde);
46        Paths { enumset, typed_enumset, core, internal, serde }
47    }
48}

50/// Generates the actual `EnumSetType` impl.
51pub fn generate_code(info: EnumSetInfo) -> SynTokenStream {
52    let paths = Paths::from(&info);
53    let name = &info.name;
54    let enumset = &paths.enumset;
55    let typed_enumset = &paths.typed_enumset;
56    let core = &paths.core;
57    let internal = &paths.internal;
58    let serde = &paths.serde;
59    let is_uninhabited = info.variants.is_empty();
60
61    let repr = match info.internal_repr() {
62        InternalRepr::U8 => quote! { u8 },
63        InternalRepr::U16 => quote! { u16 },
64        InternalRepr::U32 => quote! { u32 },
65        InternalRepr::U64 => quote! { u64 },
66        InternalRepr::U128 => quote! { u128 },
67        InternalRepr::Array(size) => quote! { #internal::ArrayRepr<{ #size }> },
68    };
69    let variant_map = info.variant_map();
70    let all_variants = match info.internal_repr() {
71        InternalRepr::U8 | InternalRepr::U16 | InternalRepr::U32 | InternalRepr::U64 => {
72            let lit = Literal::u64_unsuffixed(variant_map[0]);
73            quote! { #lit }
74        }
75        InternalRepr::U128 => {
76            let lit = Literal::u128_unsuffixed(
77                variant_map[0] as u128 | variant_map.get(1).map_or(0, |x| (*x as u128) << 64),
78            );
79            quote! { #lit }
80        }
81        InternalRepr::Array(size) => {
82            let mut new = Vec::new();
83            for i in 0..size {
84                new.push(Literal::u64_unsuffixed(*variant_map.get(i).unwrap_or(&0)));
85            }
86            quote! { #internal::ArrayRepr::<{ #size }>([#(#new,)*]) }
87        }
88    };
89
90    //
91    // Implements operator overloading on the enum type.
92    //
93    let impl_ops = if info.no_ops {
94        quote! {}
95    } else {
96        quote! {
97            #[automatically_derived]
98            impl<O: Into<#typed_enumset>> #core::ops::Sub<O> for #name {
99                type Output = #typed_enumset;
100                fn sub(self, other: O) -> Self::Output {
101                    #enumset::EnumSet::only(self) - other.into()
102                }
103            }
104            #[automatically_derived]
105            impl<O: Into<#typed_enumset>> #core::ops::BitAnd<O> for #name {
106                type Output = #typed_enumset;
107                fn bitand(self, other: O) -> Self::Output {
108                    #enumset::EnumSet::only(self) & other.into()
109                }
110            }
111            #[automatically_derived]
112            impl<O: Into<#typed_enumset>> #core::ops::BitOr<O> for #name {
113                type Output = #typed_enumset;
114                fn bitor(self, other: O) -> Self::Output {
115                    #enumset::EnumSet::only(self) | other.into()
116                }
117            }
118            #[automatically_derived]
119            impl<O: Into<#typed_enumset>> #core::ops::BitXor<O> for #name {
120                type Output = #typed_enumset;
121                fn bitxor(self, other: O) -> Self::Output {
122                    #enumset::EnumSet::only(self) ^ other.into()
123                }
124            }
125            #[automatically_derived]
126            impl #core::ops::Not for #name {
127                type Output = #typed_enumset;
128                fn not(self) -> Self::Output {
129                    !#enumset::EnumSet::only(self)
130                }
131            }
132            #[automatically_derived]
133            impl #core::cmp::PartialEq<#typed_enumset> for #name {
134                fn eq(&self, other: &#typed_enumset) -> bool {
135                    #enumset::EnumSet::only(*self) == *other
136                }
137            }
138        }
139    };
140
141    //
142    // Implements serde support.
143    //
144    let serde_repr = info.serde_repr();
145    let impl_serde_ops = match serde_repr {
146        SerdeRepr::U8 | SerdeRepr::U16 | SerdeRepr::U32 | SerdeRepr::U64 | SerdeRepr::U128 => {
147            let (serialize_repr, from_fn, to_fn) = match serde_repr {
148                SerdeRepr::U8 => (quote! { u8 }, quote! { from_u8 }, quote! { to_u8 }),
149                SerdeRepr::U16 => (quote! { u16 }, quote! { from_u16 }, quote! { to_u16 }),
150                SerdeRepr::U32 => (quote! { u32 }, quote! { from_u32 }, quote! { to_u32 }),
151                SerdeRepr::U64 => (quote! { u64 }, quote! { from_u64 }, quote! { to_u64 }),
152                SerdeRepr::U128 => (quote! { u128 }, quote! { from_u128 }, quote! { to_u128 }),
153                _ => unreachable!(),
154            };
155            let check_unknown = if info.serialize_deny_unknown {
156                quote! {
157                    let all_variants_cast =
158                        <#repr as #enumset::__internal::EnumSetTypeRepr>::#to_fn(&#all_variants);
159                    if value & !all_variants_cast != 0 {
160                        use #serde::de::Error;
161                        return #core::prelude::v1::Err(
162                            D::Error::custom("enumset contains unknown bits")
163                        )
164                    }
165                }
166            } else {
167                quote! {}
168            };
169            quote! {
170                fn serialize<S: #serde::Serializer>(
171                    set: #enumset::EnumSet<#name>, ser: S,
172                ) -> #core::result::Result<S::Ok, S::Error> {
173                    let value = <#repr as #enumset::__internal::EnumSetTypeRepr>::#to_fn(
174                        &#internal::set::get(set),
175                    );
176                    #serde::Serialize::serialize(&value, ser)
177                }
178                fn deserialize<'de, D: #serde::Deserializer<'de>>(
179                    de: D,
180                ) -> #core::result::Result<#enumset::EnumSet<#name>, D::Error> {
181                    let value = <#serialize_repr as #serde::Deserialize>::deserialize(de)?;
182                    #check_unknown
183                    let value = <#repr as #enumset::__internal::EnumSetTypeRepr>::#from_fn(value);
184                    #core::prelude::v1::Ok(#internal::set::new(value & #all_variants))
185                }
186            }
187        }
188        SerdeRepr::List => {
189            let expecting_str = format!("a list of {name}");
190            quote! {
191                fn serialize<S: #serde::Serializer>(
192                    set: #enumset::EnumSet<#name>, ser: S,
193                ) -> #core::result::Result<S::Ok, S::Error> {
194                    use #serde::ser::SerializeSeq;
195                    let mut seq = ser.serialize_seq(#core::prelude::v1::Some(set.len()))?;
196                    for bit in set {
197                        seq.serialize_element(&bit)?;
198                    }
199                    seq.end()
200                }
201                fn deserialize<'de, D: #serde::Deserializer<'de>>(
202                    de: D,
203                ) -> #core::result::Result<#enumset::EnumSet<#name>, D::Error> {
204                    struct Visitor;
205                    impl <'de> #serde::de::Visitor<'de> for Visitor {
206                        type Value = #enumset::EnumSet<#name>;
207                        fn expecting(
208                            &self, formatter: &mut #core::fmt::Formatter,
209                        ) -> #core::fmt::Result {
210                            write!(formatter, #expecting_str)
211                        }
212                        fn visit_seq<A>(
213                            mut self, mut seq: A,
214                        ) -> #core::result::Result<Self::Value, A::Error> where
215                            A: #serde::de::SeqAccess<'de>
216                        {
217                            let mut accum = #enumset::EnumSet::<#name>::new();
218                            while let #core::prelude::v1::Some(val) = seq.next_element::<#name>()? {
219                                accum |= val;
220                            }
221                            #core::prelude::v1::Ok(accum)
222                        }
223                    }
224                    de.deserialize_seq(Visitor)
225                }
226            }
227        }
228        SerdeRepr::Map => {
229            let expecting_str = format!("a map from {name} to bool");
230            quote! {
231                fn serialize<S: #serde::Serializer>(
232                    set: #enumset::EnumSet<#name>, ser: S,
233                ) -> #core::result::Result<S::Ok, S::Error> {
234                    use #serde::ser::SerializeMap;
235                    let mut map = ser.serialize_map(#core::prelude::v1::Some(set.len()))?;
236                    for bit in set {
237                        map.serialize_entry(&bit, &true)?;
238                    }
239                    map.end()
240                }
241                fn deserialize<'de, D: #serde::Deserializer<'de>>(
242                    de: D,
243                ) -> #core::result::Result<#enumset::EnumSet<#name>, D::Error> {
244                    struct Visitor;
245                    impl <'de> #serde::de::Visitor<'de> for Visitor {
246                        type Value = #enumset::EnumSet<#name>;
247                        fn expecting(
248                            &self, formatter: &mut #core::fmt::Formatter,
249                        ) -> #core::fmt::Result {
250                            write!(formatter, #expecting_str)
251                        }
252                        fn visit_map<A>(
253                            mut self, mut map: A,
254                        ) -> #core::result::Result<Self::Value, A::Error> where
255                            A: #serde::de::MapAccess<'de>
256                        {
257                            let mut accum = #enumset::EnumSet::<#name>::new();
258                            while let #core::prelude::v1::Some((val, is_present)) =
259                                map.next_entry::<#name, bool>()?
260                            {
261                                if is_present {
262                                    accum |= val;
263                                }
264                            }
265                            #core::prelude::v1::Ok(accum)
266                        }
267                    }
268                    de.deserialize_map(Visitor)
269                }
270            }
271        }
272        SerdeRepr::Array => {
273            let preferred_size = quote! {
274                <<#name as #internal::EnumSetTypePrivate>::Repr as #internal::EnumSetTypeRepr>
275                    ::PREFERRED_ARRAY_LEN
276            };
277            let (check_extra, convert_array) = if info.serialize_deny_unknown {
278                (
279                    quote! {
280                        if _val != 0 {
281                        use #serde::de::Error;
282                            return #core::prelude::v1::Err(
283                                A::Error::custom("enumset contains unknown bits")
284                            )
285                        }
286                    },
287                    quote! {
288                        use #serde::de::Error;
289                        match #enumset::EnumSet::<#name>::try_from_array(accum) {
290                            Some(x) => #core::prelude::v1::Ok(x),
291                            None => #core::prelude::v1::Err(
292                                A::Error::custom("enumset contains unknown bits")
293                            ),
294                        }
295                    },
296                )
297            } else {
298                (quote! {}, quote! {
299                    #core::prelude::v1::Ok(#enumset::EnumSet::<#name>::from_array(accum))
300                })
301            };
302            quote! {
303                fn serialize<S: #serde::Serializer>(
304                    set: #enumset::EnumSet<#name>, ser: S,
305                ) -> #core::result::Result<S::Ok, S::Error> {
306                    // read the enum as an array
307                    let array = set.as_array::<{ #preferred_size }>();
308
309                    // find the last non-zero value in the array
310                    let mut end = array.len();
311                    for i in (0..array.len()).rev() {
312                        if array[i] != 0 {
313                            break;
314                        }
315                        end = i + 1;
316                    }
317
318                    // serialize the array
319                    #serde::Serialize::serialize(&array[..end], ser)
320                }
321                fn deserialize<'de, D: #serde::Deserializer<'de>>(
322                    de: D,
323                ) -> #core::result::Result<#enumset::EnumSet<#name>, D::Error> {
324                    struct Visitor;
325                    impl <'de> #serde::de::Visitor<'de> for Visitor {
326                        type Value = #enumset::EnumSet<#name>;
327                        fn expecting(
328                            &self, formatter: &mut #core::fmt::Formatter,
329                        ) -> #core::fmt::Result {
330                            write!(formatter, "a list of u64")
331                        }
332                        fn visit_seq<A>(
333                            mut self, mut seq: A,
334                        ) -> #core::result::Result<Self::Value, A::Error> where
335                            A: #serde::de::SeqAccess<'de>
336                        {
337                            let mut accum = [0; #preferred_size];
338
339                            let mut i = 0;
340                            while let #core::prelude::v1::Some(val) = seq.next_element::<u64>()? {
341                                accum[i] = val;
342                                i += 1;
343
344                                if i == accum.len() {
345                                    break;
346                                }
347                            }
348                            while let #core::prelude::v1::Some(_val) = seq.next_element::<u64>()? {
349                                #check_extra
350                            }
351
352                            #convert_array
353                        }
354                    }
355                    de.deserialize_seq(Visitor)
356                }
357            }
358        }
359    };
360
361    //
362    // Implement the core conversion function that maps enum variants to bits.
363    //
364    let impl_internal_conversions = create_enum_conversions(&info, &paths);
365
366    //
367    // Generate the code for `Eq` and other similar basic traits.
368    //
369    let impl_basic_traits = if info.no_super_impls {
370        quote! {}
371    } else {
372        let eq_impl = if is_uninhabited {
373            quote!(#core::unreachable!(concat!(stringify!(#name), " is uninhabited.")))
374        } else {
375            quote!((*self as u32) == (*other as u32))
376        };
377
378        quote! {
379            #[automatically_derived]
380            impl #core::cmp::PartialEq for #name {
381                fn eq(&self, other: &Self) -> bool {
382                    #eq_impl
383                }
384            }
385            #[automatically_derived]
386            impl #core::cmp::Eq for #name { }
387            #[automatically_derived]
388            #[allow(clippy::expl_impl_clone_on_copy)]
389            impl #core::clone::Clone for #name {
390                fn clone(&self) -> Self {
391                    *self
392                }
393            }
394            #[automatically_derived]
395            impl #core::marker::Copy for #name { }
396        }
397    };
398
399    //
400    // Generate the code for the explicit repr trait.
401    //
402    let impl_with_repr = if info.has_explicit_repr() {
403        quote! {
404            #[automatically_derived]
405            unsafe impl #enumset::EnumSetTypeWithRepr for #name {
406                type Repr = #repr;
407            }
408        }
409    } else {
410        quote! {}
411    };
412
413    //
414    // Generate the code for compile-time operations.
415    //
416    let impl_const_opers = create_enum_const_opers(&info, &paths, &repr);
417
418    //
419    // Generate the actual enumset trait.
420    //
421    let mut generated_warnings = SynTokenStream::new();
422    for (span, warning) in &info.warnings {
423        generated_warnings.extend(quote_spanned! {
424            *span => {
425                #[deprecated(note = #warning)]
426                #[allow(non_upper_case_globals)]
427                const _w: () = ();
428                let _ = _w;
429            }
430        });
431    }
432
433    let bit_width = info.bit_width();
434    let variant_count = info.variants.len() as u32;
435    let enum_repr = info.enum_repr();
436    let enum_discrim: Vec<_> = info.variants.iter().map(|x| x.discriminant).collect();
437    let enum_names: Vec<_> = info.variants.iter().map(|x| &x.name).collect();
438    quote! {
439        const _: () = {
440            // Double check fundamental assumptions baked into a lot of the code here.
441            #core::assert!(#core::mem::size_of::<#name>() <= #core::mem::size_of::<#enum_repr>());
442            #(#core::assert!(#enum_discrim == (#name::#enum_names as i64));)*
443
444            #[automatically_derived]
445            unsafe impl #internal::EnumSetTypePrivate for #name {
446                type Repr = #repr;
447                const ALL_BITS: Self::Repr = #all_variants;
448                const BIT_WIDTH: u32 = #bit_width;
449                const VARIANT_COUNT: u32 = #variant_count;
450
451                #impl_internal_conversions
452
453                #internal::__if_serde! {
454                    #impl_serde_ops
455                }
456            }
457
458            #[automatically_derived]
459            unsafe impl #enumset::EnumSetType for #name { }
460
461            #impl_ops
462            #impl_with_repr
463            #impl_basic_traits
464            #impl_const_opers
465
466            fn __enumset_derive__generated_warnings() {
467                #generated_warnings
468            }
469        };
470    }
471}

473fn create_enum_conversions(info: &EnumSetInfo, paths: &Paths) -> SynTokenStream {
474    let name = &info.name;
475    let core = &paths.core;
476    let enum_repr = info.enum_repr();
477
478    let hint_is_transmute = quote! {
479        #[cfg(target_endian = "little")]
480        let r = *(&r as *const #enum_repr as *const #name);
481
482        #[cfg(target_endian = "big")]
483        let r = {
484            let offset =
485                #core::mem::size_of::<#enum_repr>() - #core::mem::size_of::<#name>();
486            let r = r << ((offset as #enum_repr) * 8);
487            (&r as *const #enum_repr as *const u8 as *const #name)
488        };
489    };
490
491    let is_zst = info.variants.len() == 1;
492    if info.variants.is_empty() {
493        quote! {
494            fn enum_into_u32(self) -> u32 {
495                #core::panic!(concat!(stringify!(#name), " is uninhabited."))
496            }
497            unsafe fn enum_from_u32(val: u32) -> Self {
498                #core::panic!(concat!(stringify!(#name), " is uninhabited."))
499            }
500        }
501    } else if is_zst {
502        let variant = &info.variants[0].name;
503        quote! {
504            fn enum_into_u32(self) -> u32 {
505                self as u32
506            }
507            unsafe fn enum_from_u32(val: u32) -> Self {
508                #name::#variant
509            }
510        }
511    } else if info.uses_lsb_encoding() || info.uses_msb_encoding().is_some() {
512        // Prepares output for MSB mode.
513        let process_output = if let Some(msb_repr) = info.uses_msb_encoding() {
514            quote! {
515                let r = (#msb_repr as #enum_repr) - 1 - r;
516            }
517        } else {
518            quote!()
519        };
520
521        let (names, values) = if info.bit_width() >= 1000 {
522            // This worsens codegen quality, but avoids creating big tables for no good reason.
523            let variant_bits: Vec<_> = info.variants.iter().map(|x| x.variant_bit).collect();
524            let variant_names: Vec<_> = info
525                .variants
526                .iter()
527                .map(|x| {
528                    let variant_name = &x.name;
529                    quote! { #name::#variant_name }
530                })
531                .collect();
532            (variant_names, variant_bits)
533        } else {
534            // Build a table of branches.
535            let mut names = Vec::new();
536            for i in 0..info.bit_width() {
537                // We create invalid transmutes for invalid branches that will never happen.
538                // While not very safe, this encourages the compiler to generate a transmute.
539                names.push(quote! {{
540                    let r = #i as #enum_repr;
541                    #process_output
542                    #hint_is_transmute
543                    r
544                }});
545            }
546
547            // Fill variants into the table.
548            for variant in &info.variants {
549                let variant_name = &variant.name;
550                names[variant.variant_bit as usize] = quote! { #name::#variant_name };
551            }
552
553            let values: Vec<u32> = (0..(names.len() as u32)).collect();
554
555            (names, values)
556        };
557
558        quote! {
559            #[inline]
560            fn enum_into_u32(self) -> u32 {
561                let r = self as #enum_repr;
562                #process_output
563                r as u32
564            }
565            unsafe fn enum_from_u32(val: u32) -> Self {
566                // We put these in const fields so the branches they guard aren't generated even
567                // on -O0
568                match val {
569                    // Every valid variant value has an explicit branch. If they get optimized out,
570                    // great. Otherwise, oh well, at least it's safe.
571                    #(#values => #names,)*
572                    // Default case.
573                    _ => #core::hint::unreachable_unchecked(),
574                }
575            }
576        }
577    } else if info.uses_mask_encoding() {
578        let discriminants: Vec<_> = info
579            .variants
580            .iter()
581            .map(|x| Lit::Int(LitInt::new(x.discriminant.to_string().as_str(), Span::call_site())))
582            .collect();
583        let variant_names: Vec<_> = info.variants.iter().map(|x| x.name.clone()).collect();
584        quote! {
585            fn enum_into_u32(self) -> u32 {
586                (self as #enum_repr).trailing_zeros()
587            }
588            unsafe fn enum_from_u32(val: u32) -> Self {
589                let val: #enum_repr = 1 << (val as #enum_repr);
590                match val {
591                    // Every valid variant value has an explicit branch. If they get optimized out,
592                    // great. Otherwise, oh well, at least it's safe.
593                    #(#discriminants => #name::#variant_names,)*
594                    // Default case, hints to the compiler that this is a transmute.
595                    r => {
596                        let r = r as #enum_repr;
597                        #hint_is_transmute
598                        r
599                    },
600                }
601            }
602        }
603    } else if info.uses_compact_encoding() {
604        let variant_bits: Vec<_> = info.variants.iter().map(|x| x.variant_bit).collect();
605        let variant_names: Vec<_> = info.variants.iter().map(|x| x.name.clone()).collect();
606        quote! {
607            fn enum_into_u32(self) -> u32 {
608                match self {
609                    #(#name::#variant_names => #variant_bits,)*
610                }
611            }
612            unsafe fn enum_from_u32(val: u32) -> Self {
613                match val {
614                    #(#variant_bits => #name::#variant_names,)*
615                    _ => #core::hint::unreachable_unchecked(),
616                }
617            }
618        }
619    } else {
620        panic!("Unknown encoding?");
621    }
622}

624fn create_enum_const_opers(
625    info: &EnumSetInfo,
626    paths: &Paths,
627    repr: &SynTokenStream,
628) -> SynTokenStream {
629    let name = &info.name;
630    let enumset = &paths.enumset;
631    let internal = &paths.internal;
632    let vis = &info.vis;
633
634    let value_to_bit = if info.variants.is_empty() {
635        quote! { 0 }
636    } else if info.uses_lsb_encoding() {
637        quote! { value as u32 }
638    } else if let Some(size) = info.uses_msb_encoding() {
639        quote! { ((#size as u32) - 1 - (value as u32)) }
640    } else if info.uses_mask_encoding() {
641        quote! { (value as i64).trailing_zeros() }
642    } else {
643        let variant_bits: Vec<_> = info.variants.iter().map(|x| x.variant_bit).collect();
644        let variant_names: Vec<_> = info.variants.iter().map(|x| x.name.clone()).collect();
645        quote! {
646            match value {
647                #(#name::#variant_names => #variant_bits,)*
648            }
649        }
650    };
651
652    let const_impls = match info.internal_repr() {
653        InternalRepr::U8
654        | InternalRepr::U16
655        | InternalRepr::U32
656        | InternalRepr::U64
657        | InternalRepr::U128 => {
658            quote! {
659                #[automatically_derived]
660                #[doc(hidden)]
661                impl __EnumSetInitHelper {
662                    pub const fn const_only(&self, value: #name) -> #enumset::EnumSet<#name> {
663                        #internal::set::new(1 << (#value_to_bit as #repr))
664                    }
665                }
666
667                #[automatically_derived]
668                #[doc(hidden)]
669                impl __EnumSetOpHelper {
670                    pub const fn const_union(
671                        &self,
672                        a: #enumset::EnumSet<#name>,
673                        b: #enumset::EnumSet<#name>,
674                    ) -> #enumset::EnumSet<#name> {
675                        #internal::set::new(#internal::set::get(a) | #internal::set::get(b))
676                    }
677
678                    pub const fn const_intersection(
679                        &self,
680                        a: #enumset::EnumSet<#name>,
681                        b: #enumset::EnumSet<#name>,
682                    ) -> #enumset::EnumSet<#name> {
683                        #internal::set::new(#internal::set::get(a) & #internal::set::get(b))
684                    }
685
686                    pub const fn const_symmetric_difference(
687                        &self,
688                        a: #enumset::EnumSet<#name>,
689                        b: #enumset::EnumSet<#name>,
690                    ) -> #enumset::EnumSet<#name> {
691                        #internal::set::new(#internal::set::get(a) ^ #internal::set::get(b))
692                    }
693
694                    pub const fn const_complement(
695                        &self,
696                        a: #enumset::EnumSet<#name>,
697                    ) -> #enumset::EnumSet<#name> {
698                        let mut all = #enumset::EnumSet::<#name>::all();
699                        #internal::set::new(!#internal::set::get(a) & #internal::set::get(all))
700                    }
701                }
702            }
703        }
704        InternalRepr::Array(size) => {
705            quote! {
706                #[automatically_derived]
707                #[doc(hidden)]
708                impl __EnumSetInitHelper {
709                    pub const fn const_only(&self, value: #name) -> #enumset::EnumSet<#name> {
710                        let mut repr = #internal::ArrayRepr::<{ #size }>([0; #size]);
711                        let bit = #value_to_bit;
712                        let (idx, bit) = (bit as usize / 64, bit % 64);
713                        repr.0[idx] |= 1u64 << bit;
714                        #internal::set::new(repr)
715                    }
716                }
717
718                #[automatically_derived]
719                #[doc(hidden)]
720                impl __EnumSetOpHelper {
721                    pub const fn const_union(
722                        &self,
723                        chain_a: #enumset::EnumSet<#name>,
724                        chain_b: #enumset::EnumSet<#name>,
725                    ) -> #enumset::EnumSet<#name> {
726                        let mut a = #internal::set::get(chain_a);
727                        let b = #internal::set::get(chain_b);
728                        let mut i = 0;
729                        while i < #size {
730                            a.0[i] |= b.0[i];
731                            i += 1;
732                        }
733                        #internal::set::new(a)
734                    }
735
736                    pub const fn const_intersection(
737                        &self,
738                        chain_a: #enumset::EnumSet<#name>,
739                        chain_b: #enumset::EnumSet<#name>,
740                    ) -> #enumset::EnumSet<#name> {
741                        let mut a = #internal::set::get(chain_a);
742                        let b = #internal::set::get(chain_b);
743                        let mut i = 0;
744                        while i < #size {
745                            a.0[i] &= b.0[i];
746                            i += 1;
747                        }
748                        #internal::set::new(a)
749                    }
750
751                    pub const fn const_symmetric_difference(
752                        &self,
753                        chain_a: #enumset::EnumSet<#name>,
754                        chain_b: #enumset::EnumSet<#name>,
755                    ) -> #enumset::EnumSet<#name> {
756                        let mut a = #internal::set::get(chain_a);
757                        let b = #internal::set::get(chain_b);
758                        let mut i = 0;
759                        while i < #size {
760                            a.0[i] ^= b.0[i];
761                            i += 1;
762                        }
763                        #internal::set::new(a)
764                    }
765
766                    pub const fn const_complement(
767                        &self,
768                        chain_a: #enumset::EnumSet<#name>,
769                    ) -> #enumset::EnumSet<#name> {
770                        let mut a = #internal::set::get(chain_a);
771                        let mut all = #internal::set::get(#enumset::EnumSet::<#name>::all());
772                        let mut i = 0;
773                        while i < #size {
774                            a.0[i] = !a.0[i] & all.0[i];
775                            i += 1;
776                        }
777                        #internal::set::new(a)
778                    }
779                }
780            }
781        }
782    };
783
784    quote! {
785        #[automatically_derived]
786        #[doc(hidden)]
787        #vis struct __EnumSetInitHelper;
788
789        #[automatically_derived]
790        #[doc(hidden)]
791        #vis struct __EnumSetOpHelper;
792
793        #const_impls
794
795        #[automatically_derived]
796        unsafe impl #internal::EnumSetConstHelper for #name {
797            type ConstInitHelper = __EnumSetInitHelper;
798            const CONST_INIT_HELPER: __EnumSetInitHelper = __EnumSetInitHelper;
799            type ConstOpHelper = __EnumSetOpHelper;
800            const CONST_OP_HELPER: __EnumSetOpHelper = __EnumSetOpHelper;
801        }
802    }
