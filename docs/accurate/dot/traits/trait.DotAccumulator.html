§Examples
use accurate::traits::*;
use accurate::dot::Dot2;

let x = vec![1.0, 2.0, 3.0];
let y = x.clone();

let d = Dot2::zero().absorb(x.into_iter().zip(y.into_iter()));
assert_eq!(14.0f64, d.dot())
